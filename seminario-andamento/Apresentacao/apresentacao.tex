\section{Introdução}
\frame{
    \frametitle{Introdução}
    \begin{itemize}
		\justifying
		\item \textit{MapReduce}.
		\item \textit{Framework} mais conhecido: Hadoop [1].
		\item Arquitetura utilizada: \textit{cluster}.
		\item Principal objetivo: processar grandes volumes de dados, em geral textuais.
    \end{itemize}

	\footnotetext[1]{\tiny Apache, Apache hadoop. http://hadoop.apache.org/docs/r2.6.0/index.html.}
}

\frame{
    \frametitle{Características}
    \justifying
    Características do Hadoop decorrentes do projeto:
    
	\begin{itemize}
		\justifying
		\item projetado para \textit{clusters} \textbf{homogêneos};
		\item configuração feita por arquivos de texto com valores \textit{default};
		\item arquitetura mestre e escravo;
		\item separa as atribuições em dois grandes módulos, HDFS (armazenamento) e YARN (processamento);
		\item \textbf{não possui capacidade de adaptação a ambientes levemente dinâmicos.}
	\end{itemize}
	
	
}

\frame{
	\frametitle{Objetivo}
	Objetivo Principal
	\begin{itemize}
		\item Introduzir sensibilidade ao contexto nos mecanismos de
escalonamento do Apache Hadoop através da coleta e transmissão de dados.
	\end{itemize}
	
}

\section{Desenvolvimento}
\frame{
	\frametitle{Etapas propostas}
	\begin{enumerate}
		\justifying
		\item Inserção de um coletor de contexto.
		\item Criação de uma comunicação entre os nós para melhor gestão dos recursos disponíveis.
	\end{enumerate}
}

\frame{
	\frametitle{Coletor de contexto}
	\begin{itemize}
		\item Desenvolvido para o projeto PER-MARE [4]
		\item Baseado na API padrão de monitoramento do Java [5]
		\item Novos coletores podem ser integrados para mais recursos.
	\end{itemize}

	\footnotetext[4]{\tiny STIC-AmSud, PER-MARE project. http://cosy.univ-reims.fr/PER-MARE}
	\footnotetext[5]{\tiny Oracle, Overview of java se monitoring and management.  http://docs.oracle.com/javase/7/docs/technotes/guides/management/overview.html}
}

\frame{
	\frametitle{Comunicação entre nós}
	\begin{itemize}
		\item Recursos de cada nó nunca são atualizados no escalonador.
		\item Problemas inerentes da comunicação distribuída.
		\item Solução: ZooKeeper.
	\end{itemize}	
}

\section{Experimentos e Resultados Parciais}
\frame{
	\frametitle{Configuração do ambiente}
	\begin{itemize}
		\justifying
			\item Cluster genepi, Grid'5000.
			\item 1 mestre, 4 escravos;
			\item Cada nó possuindo: 2 CPUs Intel(R) Xeon(R) E5420 2.50 GHz (totalizando 8 cores por nó) e 8 GB de memória RAM;
			\item Sistema operacional Ubuntu-x64-12.04, JDK 1.8 e a versão 2.6.0 do Hadoop.
			\item Aplicação: TeraSort em um conjunto de dados de 15GB.
	\end{itemize}
}

\frame{
	\frametitle{Configuração dos experimentos}
	Quatro casos:
	\begin{enumerate}[A)]
		\justifying
		\item Caso perfeito, recursos informados 100\% e recursos disponíveis 100\%.
		\item Comportamento \textit{default} num ambiente dinâmico, recursos informados 100\% e recursos disponíveis 50\%.
		\item Atualização ocorre antes do início do \textit{job}. Recursos disponíveis 50\% e recursos informados 50\%.
		\item Atualização ocorre após o início do \textit{job}. \textbf{Inicialização do \textit{job}:} recursos disponíveis 50\% e recursos informados 100\%. \textbf{Durante o \textit{job}:} recursos disponíveis 50\% e recursos informados 50\%. 
	\end{enumerate}
}
